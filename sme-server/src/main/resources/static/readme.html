五、完整建表语句

1. 系统用户表（sys_user）

CREATE TABLE IF NOT EXISTS sys_user (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID（自增）',
    username VARCHAR(50) NOT NULL COMMENT '用户账号（登录用）',
    password VARCHAR(100) NOT NULL COMMENT '密码（BCrypt加密存储）',
    real_name VARCHAR(50) NOT NULL COMMENT '真实姓名',
    phone VARCHAR(20) DEFAULT '' COMMENT '手机号（可做唯一校验）',
    dept_id BIGINT UNSIGNED NOT NULL COMMENT '所属部门ID，关联sys_dept表',
    role_id BIGINT UNSIGNED DEFAULT 0 COMMENT '角色ID（关联角色表，可扩展sys_role）',
    status TINYINT NOT NULL DEFAULT 1 COMMENT '状态：0-禁用 / 1-启用',
    create_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    del_flag TINYINT NOT NULL DEFAULT 0 COMMENT '逻辑删除：0-未删 / 1-已删',
    PRIMARY KEY (id),
    UNIQUE KEY uk_username (username),
    KEY idx_dept_id (dept_id),
    KEY idx_status_del (status, del_flag)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统用户表';

2. 系统部门表（sys_dept）

CREATE TABLE IF NOT EXISTS sys_dept (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    dept_name VARCHAR(50) NOT NULL COMMENT '部门名称',
    dept_code VARCHAR(50) NOT NULL COMMENT '部门编码',
    parent_id BIGINT UNSIGNED NOT NULL DEFAULT 0 COMMENT '上级部门ID（0为顶级）',
    leader VARCHAR(50) DEFAULT '' COMMENT '部门负责人',
    phone VARCHAR(20) DEFAULT '' COMMENT '部门联系电话',
    status TINYINT NOT NULL DEFAULT 1 COMMENT '状态：0-禁用 / 1-启用',
    create_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    del_flag TINYINT NOT NULL DEFAULT 0 COMMENT '逻辑删除：0-未删 / 1-已删',
    PRIMARY KEY (id),
    UNIQUE KEY uk_dept_code (dept_code),
    KEY idx_parent_id (parent_id),
    KEY idx_status_del (status, del_flag)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统部门表（含乡镇）';

3. 系统字典主表（sys_dict）

CREATE TABLE IF NOT EXISTS sys_dict (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    dict_code VARCHAR(50) NOT NULL COMMENT '字典编码（唯一）',
    dict_name VARCHAR(50) NOT NULL COMMENT '字典名称',
    sort INT NOT NULL DEFAULT 0 COMMENT '排序',
    status TINYINT NOT NULL DEFAULT 1 COMMENT '状态：0-禁用 / 1-启用',
    create_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    del_flag TINYINT NOT NULL DEFAULT 0 COMMENT '逻辑删除：0-未删 / 1-已删',
    PRIMARY KEY (id),
    UNIQUE KEY uk_dict_code (dict_code),
    KEY idx_status_del (status, del_flag)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统字典主表';

4. 系统字典项表（sys_dict_item）

CREATE TABLE IF NOT EXISTS sys_dict_item (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    dict_id BIGINT UNSIGNED NOT NULL COMMENT '关联字典主表sys_dict.id',
    item_code VARCHAR(50) NOT NULL COMMENT '项编码',
    item_name VARCHAR(50) NOT NULL COMMENT '项名称',
    sort INT NOT NULL DEFAULT 0 COMMENT '排序',
    status TINYINT NOT NULL DEFAULT 1 COMMENT '状态：0-禁用 / 1-启用',
    create_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    del_flag TINYINT NOT NULL DEFAULT 0 COMMENT '逻辑删除：0-未删 / 1-已删',
    PRIMARY KEY (id),
    KEY idx_dict_id (dict_id),
    KEY idx_status_del (status, del_flag)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统字典项表';

5. 中小微企业表（sme_enterprise）

CREATE TABLE IF NOT EXISTS sme_enterprise (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    enterprise_name VARCHAR(100) NOT NULL COMMENT '企业名称',
    credit_code VARCHAR(20) NOT NULL COMMENT '统一社会信用代码',
    enterprise_type BIGINT UNSIGNED NOT NULL COMMENT '企业类型（关联sys_dict_item，字典编码enterprise_type）',
    business_addr VARCHAR(255) DEFAULT '' COMMENT '经营地址',
    legal_person VARCHAR(50) DEFAULT '' COMMENT '法定代表人',
    phone VARCHAR(20) DEFAULT '' COMMENT '企业联系电话',
    reg_capital DECIMAL(18,2) DEFAULT 0.00 COMMENT '注册资本（万元）',
    establish_time DATE COMMENT '成立时间',
    town_id BIGINT UNSIGNED NOT NULL COMMENT '所属乡镇（关联sys_dict_item，字典编码town）',
    industry_id BIGINT UNSIGNED NOT NULL COMMENT '所属行业（关联sys_dict_item，字典编码industry）',
    business_status BIGINT UNSIGNED NOT NULL COMMENT '经营状态（关联sys_dict_item，字典编码business_status）',
    create_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    del_flag TINYINT NOT NULL DEFAULT 0 COMMENT '逻辑删除：0-未删 / 1-已删',
    PRIMARY KEY (id),
    UNIQUE KEY uk_enterprise_name (enterprise_name),
    UNIQUE KEY uk_credit_code (credit_code),
    KEY idx_enterprise_type (enterprise_type),
    KEY idx_town_id (town_id),
    KEY idx_industry_id (industry_id),
    KEY idx_business_status (business_status),
    KEY idx_del_flag (del_flag)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='中小微企业表';

6. 包抓联服务表（sme_package_service）

CREATE TABLE IF NOT EXISTS sme_package_service (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID（自增）',
    package_no VARCHAR(50) DEFAULT '' COMMENT '问题编号（系统自动生成）',
    dept_id BIGINT UNSIGNED NOT NULL COMMENT '包联部门ID，关联sys_dept.id',
    leader_item_code VARCHAR(50) NOT NULL COMMENT '包联领导编码，关联sys_dict_item.item_code',
    enterprise_id BIGINT UNSIGNED NOT NULL COMMENT '包联企业ID，关联sme_enterprise.id',
    enterprise_problem TEXT NOT NULL COMMENT '包联企业反映的问题',
    problem_receive_time DATETIME NOT NULL COMMENT '问题接收时间',
    problem_type VARCHAR(50) NOT NULL COMMENT '问题类型，关联sys_dict_item.item_code',
    handle_result VARCHAR(50) NOT NULL COMMENT '办理结果，关联sys_dict_item.item_code',
    complete_time DATETIME COMMENT '办结时间（未办结为NULL）',
    handle_remark TEXT COMMENT '办理详细说明',
    create_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    del_flag TINYINT NOT NULL DEFAULT 0 COMMENT '逻辑删除：0-未删 / 1-已删',
    PRIMARY KEY (id),
    UNIQUE KEY uk_package_no (package_no),
    KEY idx_dept_id (dept_id),
    KEY idx_leader_item_code (leader_item_code),
    KEY idx_enterprise_id (enterprise_id),
    KEY idx_problem_type (problem_type),
    KEY idx_handle_result (handle_result),
    KEY idx_problem_receive_time (problem_receive_time),
    KEY idx_del_flag (del_flag)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='企业包抓联服务表';

7. 政策发布表（sme_policy）

CREATE TABLE IF NOT EXISTS sme_policy (

七、使用说明

  1. 执行顺序：先执行建库语句，再按「基础表→业务表→发布表」顺序执行建表语句，最后执行模拟数据；
  2. 密码说明：模拟用户的密码统一为 123456，BCrypt 加密后存储，可直接用于系统登录测试；
  3. 关联说明：所有模拟数据的表关联字段（如 dept_id / enterprise_id / 字典 ID）均通过子查询自动获取，无需手动修改 ID 值；
  4. 扩展说明：可根据广河县实际部门、乡镇、企业、领导名单扩展模拟数据，仅需按现有格式补充即可；

测试查询 SQL：全关联查询

SELECT
    b.package_no,
    d.dept_name AS 包联部门,
    l.item_name AS 包联领导,
    e.enterprise_name AS 包联企业,
    b.enterprise_problem AS 反映问题,
    pt.item_name AS 问题类型,
    hr.item_name AS 办理结果,
    b.problem_receive_time AS 接收时间,
    b.complete_time AS 办结时间
FROM sme_package_service b
LEFT JOIN sys_dept d ON b.dept_id = d.id
LEFT JOIN sys_dict_item l ON b.leader_item_code = l.item_code
LEFT JOIN sme_enterprise e ON b.enterprise_id = e.id
LEFT JOIN sys_dict_item pt ON b.problem_type = pt.item_code
LEFT JOIN sys_dict_item hr ON b.handle_result = hr.item_code
WHERE b.del_flag = 0;
